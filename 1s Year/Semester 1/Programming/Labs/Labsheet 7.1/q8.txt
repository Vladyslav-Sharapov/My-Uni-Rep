START

    // Declare variables
    DECLARE length AS FLOAT
    DECLARE width AS FLOAT

    // Display messages and input
    DISPLAY "Enter length: "
    INPUT length

    DISPLAY "Enter width: "
    INPUT width

    // Validation check
    IF (length <= 0 OR width <= 0) THEN
        DISPLAY "Error, number cannot be less or equal to 0"
    ELSE
        CALL calAreaRectangle(length, width)
    END IF

// ---------------------------------------------------------------------------
// FUNCTION: calAreaRectangle
// Calculates and displays the area of a rectangle
// ---------------------------------------------------------------------------
FUNCTION calAreaRectangle(t_length AS FLOAT, t_width AS FLOAT)
    DECLARE area AS FLOAT

    SET area = t_length * t_width
    DISPLAY "Area of rectangle: " + area
END FUNCTION

END
