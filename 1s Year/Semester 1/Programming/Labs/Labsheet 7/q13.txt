START
//Variables
num1 = 0;
num2 = 0;
num3 = 0;

//Display, Input
display message "Three Numbers Divisible Check" // Heading

display message "Enter first number: "
input num1

display message "Enter second number: "
input num2

display message "Enter third number: "
input num3

// IF, Else

if (num1 % num2 == 0 && num1 % num3 == 0)
{
	display message "The first number is exactly divisible by both numbers\n";
}
else if (num1 % num2 == 0 || num1 % num3 == 0)
{
	display message "The first number is exactly divisible by one of the numbers: \n";
	if (num1 % num2 == 0)
	{
		display message num1 + " % " + num2;
	}
	else
	{
		display message num1 + " % " + num3;
	}
}
else
{
	display message "The first number is not exactly divisible by either number.\n";
}
