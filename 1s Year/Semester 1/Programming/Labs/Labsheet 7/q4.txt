START

num1 = 0;
num2 = 0;

display message "Enter the first number: ";
input num1

display message "Enter the second number: ";
input num2

if (num2 == 0)
	display message "Error. Cannot divide by zero"
else if (num1 % num2 == 0)
	display message "The first number is exactly divisible by the second number"
else
	display message "The first number is NOT exactly divisible by the second number"
end if
