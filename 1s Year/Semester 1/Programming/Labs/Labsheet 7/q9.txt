START

//Variables
length = 0;
width = 0;
perimiter = 0;

//Display & Input
display message "Perimiter of Rectangle";

display message "Enter length of the rectangle: ";
input length

display message "Enter width of the rectangle: ";
input width;

//If, Else
if(length <= 0 || width <= 0)
	display message "Error. Number cannot be less or equal to zero";
else
	perimiter =  2 * (width + length);
	display message "Rectangle perimiter: " + perimiter;

END
